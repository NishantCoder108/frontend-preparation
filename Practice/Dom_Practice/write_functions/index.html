Write four functions that add, subtract, multiply and divide two numbers? Print the result

//Add Function 

function add(a,b){
return a + b;
}


//Subract Function

function subract(){

    return a - b;
}


//Multiply Function

function multiply(){
    return a * b;
}



//Divide Functions

function divide(){
    return a / b;
}


//Result Show in Console;
let a = 3;
let b = 2;

console.log(add(a,b));     //5
console.log(subract(a,b))  //1
console.log(multiply(a,b)) //6
console.log(divide(a,b))   //1.5
